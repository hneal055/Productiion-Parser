"""
Migration Script: JSON Cache to SQLite Database
Converts existing budget_cache.json to SQLite database
"""

import json
import os
import sys
from datetime import datetime
import pandas as pd

# Add parent directory to path to import modules
sys.path.insert(0, os.path.dirname(os.path.abspath(__file__)))

from flask import Flask
from database_models import db, BudgetAnalysis, BudgetLineItem, init_db

def create_app():
    """Create minimal Flask app for migration"""
    app = Flask(__name__)
    app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///budget_analysis.db'
    app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False
    return app

def migrate_json_to_database(json_file='budget_cache.json', backup=True):
    """
    Migrate data from JSON cache to SQLite database
    
    Args:
        json_file: Path to JSON cache file
        backup: Whether to backup existing database
    """
    print("=" * 80)
    print("MIGRATION: JSON CACHE → SQLITE DATABASE")
    print("=" * 80)
    print()
    
    # Check if JSON file exists
    if not os.path.exists(json_file):
        print(f"❌ Error: {json_file} not found")
        print("   No data to migrate. Starting with fresh database.")
        return False
    
    # Load JSON data
    print(f"📂 Loading data from {json_file}...")
    try:
        with open(json_file, 'r') as f:
            cache_data = json.load(f)
        print(f"✅ Found {len(cache_data)} analyses in cache")
    except Exception as e:
        print(f"❌ Error loading JSON: {e}")
        return False
    
    # Create Flask app and initialize database
    print("\n🗄️  Initializing database...")
    app = create_app()
    
    # Backup existing database if it exists
    if backup and os.path.exists('budget_analysis.db'):
        backup_name = f'budget_analysis_backup_{datetime.now().strftime("%Y%m%d_%H%M%S")}.db'
        os.rename('budget_analysis.db', backup_name)
        print(f"📦 Backed up existing database to: {backup_name}")
    
    init_db(app)
    
    # Migrate each analysis
    print("\n🔄 Migrating analyses...")
    success_count = 0
    error_count = 0
    
    with app.app_context():
        for file_id, data in cache_data.items():
            try:
                # Create BudgetAnalysis record
                analysis = BudgetAnalysis(
                    id=file_id,
                    filename=data.get('filename', 'Unknown'),
                    upload_date=datetime.fromisoformat(data['timestamp']) if 'timestamp' in data else datetime.now(),
                    total_budget=data.get('total_budget', 0),
                    line_items=data.get('line_items', 0),
                    num_departments=data.get('num_departments', 0),
                    risk_level=data.get('risk_level', 'MODERATE'),
                    risk_score=data.get('risk_score', 0.0),
                    dataframe_json=json.dumps(data.get('df', {})),
                    risk_analysis_json=json.dumps(data.get('risk_analysis', {})),
                    optimizations_json=json.dumps(data.get('optimizations', [])),
                    analysis_timestamp=datetime.fromisoformat(data['timestamp']) if 'timestamp' in data else datetime.now()
                )
                
                db.session.add(analysis)
                
                # Add line items if DataFrame exists
                if 'df' in data and data['df']:
                    try:
                        # Reconstruct DataFrame
                        df = pd.DataFrame(data['df'])
                        
                        # Add each row as a line item
                        for idx, row in df.iterrows():
                            line_item = BudgetLineItem(
                                analysis_id=file_id,
                                category=str(row.get('Category', '')),
                                department=str(row.get('Department', '')),
                                description=str(row.get('Description', '')),
                                amount=float(row.get('Amount', 0)),
                                line_number=idx + 1
                            )
                            db.session.add(line_item)
                    except Exception as e:
                        print(f"   ⚠️  Warning: Could not migrate line items for {file_id}: {e}")
                
                success_count += 1
                print(f"   ✅ Migrated: {data.get('filename', file_id)}")
                
            except Exception as e:
                error_count += 1
                print(f"   ❌ Error migrating {file_id}: {e}")
                continue
        
        # Commit all changes
        try:
            db.session.commit()
            print(f"\n✅ Migration complete!")
            print(f"   Successfully migrated: {success_count}")
            print(f"   Errors: {error_count}")
        except Exception as e:
            db.session.rollback()
            print(f"\n❌ Error committing to database: {e}")
            return False
    
    # Archive JSON file
    if success_count > 0:
        archive_name = f'budget_cache_archived_{datetime.now().strftime("%Y%m%d_%H%M%S")}.json'
        os.rename(json_file, archive_name)
        print(f"\n📦 Archived JSON cache to: {archive_name}")
    
    return True

def verify_migration():
    """Verify migration was successful"""
    print("\n" + "=" * 80)
    print("VERIFICATION")
    print("=" * 80)
    
    if not os.path.exists('budget_analysis.db'):
        print("❌ Database file not found!")
        return False
    
    app = create_app()
    init_db(app)
    
    with app.app_context():
        from database_models import get_database_stats
        
        stats = get_database_stats()
        
        print(f"\n📊 Database Statistics:")
        print(f"   Total Analyses: {stats['total_analyses']}")
        print(f"   Total Line Items: {stats['total_line_items']}")
        print(f"   Total Budget Tracked: ${stats['total_budget_tracked']:,.2f}")
        print(f"   Average Budget: ${stats['avg_budget']:,.2f}")
        print(f"\n   Risk Distribution:")
        for risk_level, count in stats['risk_distribution']:
            print(f"      {risk_level}: {count}")
        
        if stats['total_analyses'] > 0:
            print("\n✅ Migration verified successfully!")
            return True
        else:
            print("\n⚠️  Database is empty!")
            return False

def show_help():
    """Show usage instructions"""
    print("""
    ============================================================================
    JSON TO DATABASE MIGRATION TOOL
    ============================================================================
    
    This script migrates your budget analysis data from JSON cache to SQLite.
    
    USAGE:
        python migrate_to_database.py [options]
    
    OPTIONS:
        --help              Show this help message
        --no-backup         Skip database backup
        --verify-only       Only verify existing database
        --json FILE         Specify custom JSON file (default: budget_cache.json)
    
    EXAMPLES:
        # Standard migration
        python migrate_to_database.py
        
        # Migration without backup
        python migrate_to_database.py --no-backup
        
        # Verify existing database
        python migrate_to_database.py --verify-only
        
        # Custom JSON file
        python migrate_to_database.py --json my_cache.json
    
    WHAT IT DOES:
        1. Backs up existing database (if any)
        2. Creates new SQLite database
        3. Migrates all analyses from JSON
        4. Archives original JSON file
        5. Verifies migration success
    
    SAFETY:
        - Original JSON is archived, not deleted
        - Existing database is backed up
        - All operations are logged
    
    ============================================================================
    """)

# ============================================================================
# MAIN SCRIPT
# ============================================================================

if __name__ == '__main__':
    import sys
    
    # Parse command line arguments
    args = sys.argv[1:]
    
    if '--help' in args or '-h' in args:
        show_help()
        sys.exit(0)
    
    if '--verify-only' in args:
        verify_migration()
        sys.exit(0)
    
    # Get options
    backup = '--no-backup' not in args
    
    json_file = 'budget_cache.json'
    if '--json' in args:
        idx = args.index('--json')
        if idx + 1 < len(args):
            json_file = args[idx + 1]
    
    # Run migration
    print("\n" + "🚀" * 40)
    print()
    
    success = migrate_json_to_database(json_file, backup)
    
    if success:
        verify_migration()
        
        print("\n" + "=" * 80)
        print("✅ MIGRATION SUCCESSFUL!")
        print("=" * 80)
        print("\nNEXT STEPS:")
        print("1. Update your Flask app to use the database")
        print("2. Test the application")
        print("3. Keep backups for safety")
        print("\n" + "🎉" * 40)
    else:
        print("\n" + "=" * 80)
        print("❌ MIGRATION FAILED")
        print("=" * 80)
        print("\nCheck the error messages above for details.")
        sys.exit(1)
