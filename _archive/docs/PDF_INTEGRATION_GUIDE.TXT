```python
@app.route('/generate-pdf/<file_id>')
def generate_pdf_report_route(file_id):
    \"\"\"Generate and download PDF report for analysis\"\"\"
    if file_id not in analysis_cache:
        flash('Analysis not found', 'error')
        return redirect(url_for('index'))
    
    try:
        from pdf_report_generator import generate_pdf_report
        
        data = analysis_cache[file_id]
        df = data['df']
        
        # Prepare budget data
        budget_data = {
            'filename': data['filename'],
            'total_budget': data['total_budget'],
            'line_items': data['line_items'],
            'num_departments': len(set(df['Department'])) if 'Department' in df.columns else 0,
            'risk_level': data['risk_level'],
            'departments': {}
        }
        
        # Get department breakdown
        if 'Department' in df.columns and 'Amount' in df.columns:
            total = df["Amount"].sum()
            for dept, group in df.groupby("Department"):
                dept_total = group["Amount"].sum()
                budget_data['departments'][dept] = {
                    "amount": float(dept_total),
                    "percentage": float(dept_total / total * 100) if total > 0 else 0,
                    "items": len(group)
                }
        
        # Get risk analysis
        risk_manager = RiskManager()
        risk_analysis = risk_manager.analyze_risks(df)
        risk_data = {
            'risk_level': risk_analysis['summary']['risk_level'],
            'overall_risk_score': risk_analysis['summary']['overall_risk_score'],
            'risk_categories': risk_analysis['items_by_category']
        }
        
        # Get optimization data
        optimizations = find_optimizations(df)
        
        # Get visualization paths (if you have the visualization function)
        visualizations = {}
        if hasattr(data, 'visualizations'):
            visualizations = data['visualizations']
        
        # Generate PDF
        pdf_filename = f"budget_report_{file_id}.pdf"
        pdf_path = os.path.join(app.config['OUTPUT_FOLDER'], pdf_filename)
        
        generate_pdf_report(budget_data, risk_data, optimizations, pdf_path, visualizations)
        
        # Serve the PDF
        return send_file(
            pdf_path, 
            as_attachment=True,
            download_name=f"{data['filename']}_report.pdf",
            mimetype='application/pdf'
        )
        
    except Exception as e:
        flash(f'Error generating PDF: {str(e)}', 'error')
        print(f"PDF Error: {str(e)}")  # Debug
        import traceback
        traceback.print_exc()
        return redirect(url_for('view_analysis', file_id=file_id))
```

🎯 ADVANCED: Add Visualizations
-------------------------------

To include charts in the PDF, add this function to web_app_enhanced.py:

```python
def create_visualizations(df, output_folder):
    \"\"\"Create visualization charts for PDF\"\"\"
    import matplotlib
    matplotlib.use('Agg')
    import matplotlib.pyplot as plt
    
    vis_paths = {}
    
    try:
        # Pie chart - Department allocation
        if 'Department' in df.columns:
            fig, ax = plt.subplots(figsize=(8, 6))
            dept_totals = df.groupby('Department')['Amount'].sum()
            ax.pie(dept_totals, labels=dept_totals.index, autopct='%1.1f%%')
            ax.set_title('Budget Allocation by Department')
            pie_path = os.path.join(output_folder, 'dept_pie.png')
            plt.savefig(pie_path, bbox_inches='tight', dpi=150)
            plt.close()
            vis_paths['pie_chart'] = pie_path
        
        # Bar chart - Top items
        fig, ax = plt.subplots(figsize=(10, 6))
        top_items = df.nlargest(10, 'Amount')
        ax.barh(range(len(top_items)), top_items['Amount'])
        ax.set_yticks(range(len(top_items)))
        ax.set_yticklabels(top_items['Description'][:20])
        ax.set_xlabel('Amount ($)')
        ax.set_title('Top 10 Budget Items')
        bar_path = os.path.join(output_folder, 'top_items_bar.png')
        plt.savefig(bar_path, bbox_inches='tight', dpi=150)
        plt.close()
        vis_paths['bar_chart'] = bar_path
        
    except Exception as e:
        print(f"Error creating visualizations: {e}")
    
    return vis_paths
```

Then in the PDF route, add:
```python
vis_paths = create_visualizations(df, app.config['OUTPUT_FOLDER'])
visualizations = {}
for key, path in vis_paths.items():
    if os.path.exists(path):
        visualizations[key] = path
```
