    <script>
        // Auth Service
        class AuthService {
            static async login(credentials) {
                // Simulate API call with delay
                return new Promise((resolve, reject) => {
                    setTimeout(() => {
                        if (credentials.username === 'admin' && credentials.password === 'AuraDemo2024!') {
                            resolve({ 
                                token: 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJ1c2VybmFtZSI6ImFkbWluIiwiaWF0IjoxNTE2MjM5MDIyfQ',
                                user: { name: 'Admin User', role: 'administrator' }
                            });
                        } else {
                            reject(new Error('Invalid credentials'));
                        }
                    }, 1500);
                });
            }
            
            static setSession(token) {
                localStorage.setItem('aura_token', token);
            }
            
            static isAuthenticated() {
                return !!localStorage.getItem('aura_token');
            }
            
            static logout() {
                localStorage.removeItem('aura_token');
            }
        }

        // Login Form Handler
        class LoginForm {
            constructor() {
                this.form = document.getElementById('login-form');
                this.username = document.getElementById('username');
                this.password = document.getElementById('password');
                this.errorElement = document.getElementById('error-message');
                this.loginButton = document.getElementById('login-button');
                this.togglePasswordBtn = document.getElementById('toggle-password');
                this.demoInfo = document.getElementById('demo-info');
                
                this.init();
            }
            
            init() {
                this.form.addEventListener('submit', this.handleSubmit.bind(this));
                this.togglePasswordBtn.addEventListener('click', this.togglePassword.bind(this));
                this.demoInfo.addEventListener('click', this.autoFillDemoCredentials.bind(this));
                
                // Check if user is already authenticated
                if (AuthService.isAuthenticated()) {
                    this.showSuccess('You are already logged in. Redirecting...');
                    setTimeout(() => {
                        window.location.href = '/dashboard';
                    }, 2000);
                }
            }
            
            async handleSubmit(e) {
                e.preventDefault();
                
                if (this.validateForm()) {
                    await this.submitForm();
                }
            }
            
            validateForm() {
                this.clearErrors();
                
                let isValid = true;
                
                if (!this.username.value.trim()) {
                    this.showFieldError(this.username, 'Username is required');
                    isValid = false;
                }
                
                if (!this.password.value) {
                    this.showFieldError(this.password, 'Password is required');
                    isValid = false;
                }
                
                return isValid;
            }
            
            async submitForm() {
                const credentials = {
                    username: this.username.value,
                    password: this.password.value
                };
                
                try {
                    this.setLoadingState(true);
                    const result = await AuthService.login(credentials);
                    AuthService.setSession(result.token);
                    this.showSuccess('Login successful! Redirecting...');
                    
                    // Redirect to dashboard after successful login
                    setTimeout(() => {
                        window.location.href = '/dashboard';
                    }, 1500);
                    
                } catch (error) {
                    this.showError(error.message);
                } finally {
                    this.setLoadingState(false);
                }
            }
            
            togglePassword() {
                const type = this.password.type === 'password' ? 'text' : 'password';
                this.password.type = type;
                this.togglePasswordBtn.textContent = type === 'password' ? 'Show' : 'Hide';
            }
            
            autoFillDemoCredentials() {
                this.username.value = 'admin';
                this.password.value = 'AuraDemo2024!';
                this.showInfo('Demo credentials filled. Click Login to continue.');
            }
            
            showError(message) {
                this.errorElement.textContent = message;
                this.errorElement.style.display = 'flex';
                this.animateError();
            }
            
            showSuccess(message) {
                this.errorElement.style.backgroundColor = 'rgba(76, 175, 80, 0.1)';
                this.errorElement.style.color = '#2e7d32';
                this.errorElement.textContent = message;
                this.errorElement.style.display = 'flex';
            }
            
            showInfo(message) {
                this.errorElement.style.backgroundColor = 'rgba(33, 150, 243, 0.1)';
                this.errorElement.style.color = '#1565c0';
                this.errorElement.textContent = message;
                this.errorElement.style.display = 'flex';
            }
            
            showFieldError(field, message) {
                field.style.borderColor = 'var(--error-color)';
                this.showError(message);
            }
            
            clearErrors() {
                this.errorElement.style.display = 'none';
                this.errorElement.style.backgroundColor = 'rgba(211, 47, 47, 0.1)';
                this.errorElement.style.color = 'var(--error-color)';
                
                // Reset field borders
                this.username.style.borderColor = '';
                this.password.style.borderColor = '';
            }
            
            setLoadingState(loading) {
                this.loginButton.disabled = loading;
                this.loginButton.textContent = loading ? 
                    '<span class="loading">Logging in...</span>' : 
                    'Login to Dashboard';
            }
            
            animateError() {
                this.errorElement.animate([
                    { transform: 'translateX(-5px)' },
                    { transform: 'translateX(5px)' },
                    { transform: 'translateX(0)' }
                ], {
                    duration: 300,
                    iterations: 2
                });
            }
        }

        // Initialize the application when DOM is loaded
        document.addEventListener('DOMContentLoaded', function() {
            new LoginForm();
            
            // Add loading animation style
            const style = document.createElement('style');
            style.textContent = `
                .loading {
                    display: inline-block;
                    width: 20px;
                    height: 20px;
                    border: 2px solid #ffffff;
                    border-radius: 50%;
                    border-top-color: transparent;
                    animation: spin 1s ease-in-out infinite;
                }
                
                @keyframes spin {
                    to { transform: rotate(360deg); }
                }
            `;
            document.head.appendChild(style);
        });

        // Mock dashboard redirection for demo purposes

        // For demo, we'll prevent actual navigation and show a message
        const originalLocation = window.location;
        window.location.href = new URL(originalLocation);
