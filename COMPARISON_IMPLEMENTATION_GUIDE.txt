    if not session_id:

    if session_id not in budget_history:
        budget_history[session_id] = []
    
    budget_history[session_id].append({
        'id': file_id,
        'name': file.filename,
        'total': total_budget,
        'items': len(df),
        'timestamp': timestamp.strftime('%B %d, %Y at %I:%M %p')
    })

    if len(budget_history[session_id]) > 10:
        budget_history[session_id] = budget_history[session_id][-10:]
    
    # Set session cookie if new
    if not request.cookies.get('session_id'):
        resp = make_response(redirect(url_for('view_analysis', file_id=file_id)))
        resp.set_cookie('session_id', session_id, max_age=30*24*60*60)  # 30 days
        return resp

🐛 TROUBLESHOOTING
════════════════════════════════════════════════════════════════════════════════

ISSUE: ImportError: No module named 'budget_comparison'
SOLUTION: Make sure budget_comparison.py is in the same directory as your Flask app

ISSUE: CSS styles not loading
SOLUTION: Check that comparison-styles.css is in static/css/ folder

ISSUE: Charts not rendering on comparison page
SOLUTION: Verify Chart.js is loading (check browser console for errors)

ISSUE: "Comparison not found" error
SOLUTION: Make sure both budgets are uploaded in the same browser session

ISSUE: Session not persisting
SOLUTION: Check that cookies are enabled in your browser

ISSUE: Budget history empty after restart
SOLUTION: This is expected - history is session-based. For persistence, you'll need database (PATH B)
