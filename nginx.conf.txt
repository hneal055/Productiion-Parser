# Simplified Dockerfile using default nginx config
FROM nginx:alpine

# Install curl for health checks
RUN apk add --no-cache curl

# Copy static files to nginx
COPY . /usr/share/nginx/html

# Create non-root user
RUN addgroup -g 1001 -S nodejs && \
    adduser -S aura -u 1001

# Set ownership
RUN chown -R aura:nodejs /usr/share/nginx/html && \
    chown -R aura:nodejs /var/cache/nginx && \
    chown -R aura:nodejs /var/log/nginx && \
    chown -R aura:nodejs /etc/nginx/conf.d

# Switch to non-root user
USER aura

# Expose port
EXPOSE 80

# Health check
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:80/ || exit 1

# Start nginx
CMD ["nginx", "-g", "daemon off;"]
